expr    : KEYWORD: VAR ID EQ expr
        : term ((PLUS | MINUS) term)*

term    : factor ((MUL | DIV) factor)*

factor  : (PLUS | MINUS) factor
        : power

power   : nachos (POW factor)*

nachos  : INT | FLOAT | ID
        : LPAREN expr RPAREN

PLUS    : '+'
MINUS   : '-'
MUL     : '*'
DIV     : '/'
